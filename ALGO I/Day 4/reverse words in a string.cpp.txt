class Solution {
public:
    string reverseWords(string s) {
        int n = s.length();
        stack<string>st;
        string temp ="";
        string ans = "";

       for(int i =0;i<n;i++)
        {
	if(s[i] == ' ')
	{
        if(temp.length() > 0)
             st.push(temp);
        temp ="";
	}
	else
	{
		temp+=s[i];
	}


ans+=temp;
        
       }
};
